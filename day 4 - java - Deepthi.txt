class Rectangle {
    private double length;
    private double width;

    // Constructor
    public Rectangle(double length, double width) {
        this.length = length;
        this.width = width;
    }

    // Getter methods
    public double getLength() {
        return length;
    }

    public double getWidth() {
        return width;
    }

    // Method to calculate area
    public double calculateArea() {
        return length * width;
    }

    // Method to calculate perimeter
    public double calculatePerimeter() {
        return 2 * (length + width);
    }
}

import java.util.Scanner;

class Bicycle {
    public int gear;
    public int speed;

    // Constructor
    public Bicycle(int gear, int speed) {
        this.gear = gear;
        this.speed = speed;
    }

    public void applyBrake(int decrement) {
        speed -= decrement;
    }

    public void speedUp(int increment) {
        speed += increment;
    }

    public String toString() {
        return "No of gears are " + gear + "\nspeed of bicycle is " + speed;
    }

    // Extra: provide a main here so platforms that try to start Bicycle will succeed
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        if (!sc.hasNextInt()) return;
        int gear = sc.nextInt();
        if (!sc.hasNextInt()) return;
        int speed = sc.nextInt();
        if (!sc.hasNextInt()) return;
        int seatHeight = sc.nextInt();

        MountainBike mb = new MountainBike(gear, speed, seatHeight);
        System.out.println(mb.toString());
    }
}

class MountainBike extends Bicycle {
    public int seatHeight;

    // Constructor
    public MountainBike(int gear, int speed, int seatHeight) {
        super(gear, speed);
        this.seatHeight = seatHeight;
    }

    public void setHeight(int newValue) {
        seatHeight = newValue;
    }

    @Override
    public String toString() {
        return super.toString() + "\nseat height is " + seatHeight;
    }
}

Question 4

Volume
Write a program in java to exploit constructor overloading. (cube and cuboid volume calculation)

i)  Create a class ThreeDimensionShape with three variables width, height, depth;

ii)  Create three constructors

a)     Doesn’t accepts any parameter

b)     Accepts 1 parameter of type double(length)

c)     Accepts 3 parameter of type double(width, height, depth)

iii)  Create a method volume of type double that returns a product of width, height, and depth.

iv)  in case of No argument constructor make length=width=depth = 0;

v)  in case of 1 argument constructor make length=width=depth = length;

Your answer

class ThreeDimensionShape {
    double width;
    double height;
    double depth;

    // No-argument constructor
    ThreeDimensionShape() {
        width = height = depth = 0;
    }

    // One-argument constructor (cube)
    ThreeDimensionShape(double side) {
        width = height = depth = side;
    }

    // Three-argument constructor (cuboid)
    ThreeDimensionShape(double width, double height, double depth) {
        this.width = width;
        this.height = height;
        this.depth = depth;
    }

    // Method to calculate volume
    double volume() {
        return width * height * depth;
    }

    // Main method (inside this class)
    public static void main(String[] args) {
        ThreeDimensionShape shape1 = new ThreeDimensionShape(10, 7, 3);
        ThreeDimensionShape shape2 = new ThreeDimensionShape();
        ThreeDimensionShape shape3 = new ThreeDimensionShape(8);

        System.out.println("Volume of shape1 is " + shape1.volume());
        System.out.println("Volume of shape2 is " + shape2.volume());
        System.out.println("Volume of shape3 is " + shape3.volume());
    }
}
